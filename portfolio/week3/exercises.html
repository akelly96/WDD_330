let mountainTable = "<table><tr>
                    ;<th>Mountain</th>
                    <th>Height</th>
                    <th>Location</th>
                    ;</tr>"
for (let i = 0; i < MOUNTAINS.length; i++) {
    mountainTable += "<tr><td>" 
                    + MOUNTAINS[i].name 
                    + "</td>" 
                    +
                    "<td>" 
                    + MOUNTAINS[i].height 
                    + "</td>" 
                    + "<td>" 
                    + MOUNTAINS[i].place 
                    + "</td></tr>";
}
mountainTable += "</table>";
document.getElementById("mountains").innerHTML = mountainTable;
}
