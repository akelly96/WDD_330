Exercise 1

class Vec {
    constructor(x,y){
        this.x = x;
        this.y = y;
    }
    
    plus(vec) {
        return new Vec(this.x + vec.x, this.y + vec.y);
    }
    
    minus(vec) {
        return new Vec(this.x - vec.x, this.y - vec.y);
    }
    
    get length() {
        return Math.sqrt(this.x * this.x + this.y * this.y); 
    }
}

Exercise 2

class Group {
    constructor() {
        this.array = [];
    }
    
    static from(arg){
        let group = new Group;
        for (var i = arg[0]; i <= arg[1]; i++) {
        group.array.push(i); 
        }
        return group;
    }
    
    has(number) {
        if (group.array.indexOf(number) !== -1) {
        return true;
        } else {
        return false;
        }    
    }
    
    add(number) {
        if (group.array.indexOf(number) === -1) {
        group.array.push(number);
        }
    }
    
    delete(number) {
        var index = group.array.indexOf(number)
        if (index !== -1) {
        group.array.splice(index, 1);
        }
    }
}
